  doRecursion = (data) => {
    data.forEach(x => {
      if (x.children) {
        this.doRecursion(x.children);
      }
      x.show = '0';
    });
  }

  doRecursionMinus = (data, item) => {
    data.forEach(x => {
      if (x.children) {
        this.doRecursionMinus(x.children, item);
      }
      x.show = '0';
    });
  }

  recursionChange = (data, item) => {
    data.forEach(x => {
      if (x.children) {
        this.recursionChange(x.children, item);
      }
      if (x.category == item) {
        x.show = '1';
      }
    });
  }

  recursionAdd = (data, item) => {
    data.forEach(x => {
      if (x.children) {
        this.recursionAdd(x.children, item);
      }
      if (x.category == item) {
        x.show = '1';
      }
    });
  }
  recursionMinus = (data, item) => {
    data.forEach(x => {

        this.recursionMinus(x.children, item);
