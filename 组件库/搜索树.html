const {TreeNode} = Tree;
let targetKeys = [];

class KindManagement extends Component {
  constructor(props) {
    super(props);
    this.state = {
      expandedKeys: [],
      searchValue: "",
      autoExpandParent: true,
      selectedTitle: undefined
    }
  }

  onSelect = (selectedKeys, info) => {
    console.log('selected', selectedKeys, info);
  };

  onCheck = (checkedKeys, info) => {
    console.log('onCheck', checkedKeys, info);
  };

  onChange = e => {
    const {value} = e.target;
    this.setState({inputValue: value});
    targetKeys = [];
    this.recursionFind(treeData, value);
    this.setState({
      expandedKeys: targetKeys,
      autoExpandParent: true
    });
  };

  recursionFind = (data, target) => {
    data.forEach(x => {
      if (x.children.length > 0) {
        this.recursionFind(x.children, target);
      }
      if (target && x.title.indexOf(target) > -1) {
        targetKeys.push(x.key);
      }
    });
  };

  onExpand = expandedKeys => {
    this.setState({
      expandedKeys,
      autoExpandParent: false
    });
  };

  generateLeftContent = () => {
    const {expandedKeys, autoExpandParent} = this.state;
    return (
      <>
        <p style={{fontWeight: 'bold'}}>合同类型</p>
        <Input.Search
          style={{marginBottom: 8}}
          placeholder="Search"
          onChange={this.onChange}
        />
        <Tree
          onExpand={this.onExpand}
          expandedKeys={expandedKeys}
          autoExpandParent={autoExpandParent}
          onSelect={this.onSelect}
        >
          {this.renderTreeNodes(treeData)}
        </Tree>
      </>
    );
  }

  renderTreeNodes = data => {
    const {inputValue} = this.state;
    if (data && data.length > 0) {
      return data.map(item => {
        if (item.children && item.children.length > 0) {
          return (
            <Tree.TreeNode title={<>{item.title}</>} key={item.key}>
              {this.renderTreeNodes(item.children)}
            </Tree.TreeNode>
          );
        }
        const index = item.title.indexOf(inputValue);
        const beforeStr = item.title.substr(0, index);
        const afterStr = item.title.substr(index + inputValue?.length);
        const title =
          index > -1 ? (
            <span>
              {beforeStr}
              <span style={{color: '#f50'}}>{inputValue}</span>
              {afterStr}
            </span>
          ) : (
            <span>{item.title}</span>
          );
        return (
          <Tree.TreeNode
            title={title}
            key={item.key}
          />
        );
      });
    }
    return "";
  };

  generateRightContent = () => {
    return (
      <NormLoanContract/>
    );
  }

  render() {
    return (
      <YRLeftRightLayout leftContent={this.generateLeftContent()} rightContent={this.generateRightContent()}/>
    );
  }
}

export default KindManagement;
